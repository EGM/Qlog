[+] qlog

Quick Logger for JavaScript

Strings

qlog( "Strings" );

"Strings"

Numbers

qlog( 123 );

123

Objects

qlog( { } );

{ }

qlog( {name:"John"} );

		{
		"name":"John"
		}

Arrays

qlog( [ ] );

[ ]

qlog( [1,3,2] );

		[
		  1,
		  3,
		  2
		]

qlog( ["piano", "lounge chair"] );

		[
		  "piano",
		  "lounge chair"
		]

qlog( [{name: "John"}, {name: "Sue"}] );

		[
		  {
		    "name":"John"
		  },
		  {
		    "name":"Sue"
		  }
		]

Null

qlog( null );

null

Undefined

var undefined_var;

qlog( undefined_var );

undefined

Function

function hello() {
   return "hello world";
} 

		function hello() {
		  return "hello world";
		}
